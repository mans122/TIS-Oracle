5. 2번테이블에 데이터가 입력되면 1번 테이블을 업데이트 하시오.
	insert into 입고 ....
	update 재고 set 재고량=현재재고량+입고량 ...
create or replace procedure p_invenin(
    p_No number,
    p_gCode inven2.gcode%type,
    p_Ea number,
    p_Date date
    )
is 
    PRAGMA AUTONOMOUS_TRANSACTION;
begin
    commit;
    insert into invenin2 values (p_No,p_gCode,p_Ea,p_date);
    update inven2 set gea =gea + p_ea where gcode = p_gcode;
    
    exception when others then
        RAISE_APPLICATION_ERROR(-20000, '에러발생');
        rollback;
end;
	

6. 3번테이블에 데이터가 입력되면 1번 테이블을 업데이트 하시오.
	insert into 출고 ....
	update 재고 set 재고량=현재재고량-출고량 ...

create or replace procedure p_invenOut(
    p_No number,
    p_gCode inven2.gcode%type,
    p_Ea number,
    p_Date date
    )
is 
    PRAGMA AUTONOMOUS_TRANSACTION;
    vn_cnt number:=0;
begin
    commit;
    select count(*) 
    into vn_cnt from inven2 
    where gcode=p_gcode and gea >= p_ea;
    
    if vn_cnt>=1 then
        insert into invenout2 values (p_No,p_gCode,p_Ea,p_date);
        update inven2 set gea = gea - p_ea where gcode = p_gcode;
        
    else
        RAISE_APPLICATION_ERROR(-20000, '출고 수량이 재고 수량보다 커서 출고할 수 없습니다');
        rollback;
    end if;
end;
